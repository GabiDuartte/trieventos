create table estabelecimentos(
	id_estabelecimentos serial primary key,
	titulo_card varchar(30),
	texto_card varchar(60),
	descricao_estabelecimento text
);

CREATE TABLE usuarios (
    id_usuario serial PRIMARY KEY,
    nome varchar(50),
    email varchar(50),
    senha varchar(50),
	tipo_usuario varchar(20)
);

create table usuarios_estabelecimentos(
	id_usuario_e serial PRIMARY KEY,
    nome varchar(50),
    email varchar(50),
    senha varchar(50),
	tipo_usuario varchar(20)
);
